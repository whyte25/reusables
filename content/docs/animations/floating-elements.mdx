Usage

```tsx
import FloatElements from "@/components/ui/floating-elements"

export default function FloatingElementsDemo() {
  return (
    <FloatElements title="Trusted by" className="mt-20">
      {/* Your floating elements here */}
    </FloatElements>
  )
}
```
