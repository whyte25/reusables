## Usage

```tsx
import SplitButton from "@/components/ui/split-button"
```

```tsx
const options = [
  { label: "Option 1", onClick: () => console.log("Option 1 clicked") },
  { label: "Option 2", onClick: () => console.log("Option 2 clicked") },
]

;<SplitButton
  options={options}
  mainAction={() => console.log("Main action clicked")}
/>
```
