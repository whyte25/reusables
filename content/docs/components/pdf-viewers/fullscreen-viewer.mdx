title: Fullscreen PDF Viewer
description: A fullscreen PDF viewer component with zoom controls and navigation.
---

## Features

- Fullscreen mode
- Zoom controls
- Page navigation
- Download option
- Dark mode support

## Usage

```tsx
import { FullscreenViewer } from "@/components/ui/pdf-viewers/fullscreen-viewer";

export default function PDFViewer() {
  return (
    <FullscreenViewer
      url="path/to/your.pdf"
      onClose={() => console.log("closed")}
    />
  );
}
```
